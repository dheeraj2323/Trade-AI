@app.route('/auth/login', methods=['POST'])
def auth_login():
    email = request.form.get('email')
    password = request.form.get('password')
    hashed = hashlib.sha256(password.encode()).hexdigest()
    conn = sqlite3.connect(DB_PATH)
    c = conn.cursor()
    user = c.execute("SELECT * FROM users WHERE email=? AND password=?", (email, hashed)).fetchone()
    conn.close()
    if user:
        session['user_id'] = user[0]
        session['user_name'] = user[1]
        session['user_color'] = user[5] if len(user) > 5 else '#1D9E75'
        return redirect('/')
    return redirect('/login?error=Invalid email or password. Please try again.')

@app.route('/auth/logout')
def auth_logout():
    session.clear()
    return redirect('/login')
