   def watchlist():
       uid = session.get('user_id', None)
       if not uid:
           return render_template('watchlist.html', 
               watchlist=[], 
               all_stocks=list(STOCKS.keys()),
               logged_in=False)
       conn = sqlite3.connect(DB_PATH)
       rows = conn.execute(
           "SELECT symbol FROM user_watchlist WHERE user_id=?", (uid,)
       ).fetchall()
       conn.close()
       user_stocks = [r[0] for r in rows]

           watchlist=user_stocks,
           all_stocks=list(STOCKS.keys()),
